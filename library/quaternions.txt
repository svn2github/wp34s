This is a basic suite of quaternions operations for the WP 34S.
Original code by Jean-Marc Baillard from:

	http://hp41programs.yolasite.com/quaternions.php


LBL'Q+Q'
	cpx RCL+ A
	cpx Rv
	cpx RCL+ A
	cpx R^
	RTN
LBL'Q-Q'
	cpx RCL- A
	cpx +/-
	cpx Rv
	cpx RCL- A
	cpx +/-
	cpx R^
	RTN

LBL'Q^Q'
	cpx Rv
	cpx Rv
	XEQ 97
	XEQ 95
	GTO 96

LBL'LNQ'
LBL 97
	DEG
	XEQ 99
	LN
	Rv
	D->R
	->RECT
	Rv
	->RECT
	cpx R^
	RTN

LBL'POL'
LBL 99
	cpx Rv
	->POL
	R^
	->POL
	R^
	->POL
	RTN

LBL'e^Q'
LBL 96
	DEG
	cpx Rv
	->POL
	R^
	->POL
	R->D
	R^
	e^x
LBL'REC'
LBL 98
	->RECT
	Rv
	->RECT
	Rv
	->RECT
	cpx R^
	RTN


LBL'Q[times]Q'
LBL 95
	RTN

LBL'Q^R'
	RTN
LBL'Q^Q'
	RtN
